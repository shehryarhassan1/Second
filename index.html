Python Standard Libraries Cheat Sheet (Python 3.12+)

Text Processing

string

Use Case: Constants, utilities, and common string operations.

import string
print(string.ascii_letters)  # All letters
print(string.digits)         # '0123456789'

re

Use Case: Regular expressions for pattern matching.

import re
pattern = r"\d+"
text = "There are 42 apples"
print(re.findall(pattern, text))  # ['42']

difflib

Use Case: Compare sequences or find differences between strings.

from difflib import SequenceMatcher
s1 = "Python"
s2 = "Pyhton"
print(SequenceMatcher(None, s1, s2).ratio())  # similarity ratio

textwrap

Use Case: Wrap and format text blocks.

import textwrap
text = "Python is amazing for programming."
print(textwrap.fill(text, width=10))

unicodedata

Use Case: Working with Unicode characters.

import unicodedata
print(unicodedata.name('A'))  # 'LATIN CAPITAL LETTER A'

Data Types & Algorithms

collections

Use Case: Efficient data structures like deque, Counter, OrderedDict.

from collections import Counter
data = [1,2,2,3,3,3]
print(Counter(data))  # Counter({3: 3, 2: 2, 1: 1})

heapq

Use Case: Heap queue algorithms (priority queues).

import heapq
nums = [5,1,7,3]
heapq.heapify(nums)
print(heapq.heappop(nums))  # 1

bisect

Use Case: Binary search and insert into sorted lists.

import bisect
lst = [1,3,5,7]
bisect.insort(lst, 4)
print(lst)  # [1, 3, 4, 5, 7]

array

Use Case: Typed numeric arrays for efficient storage.

import array
arr = array.array('i', [1,2,3])
print(arr[0])  # 1

dataclasses

Use Case: Simple class with auto-generated methods.

from dataclasses import dataclass
@dataclass
class Point:
    x: int
    y: int

p = Point(1,2)
print(p)  # Point(x=1, y=2)

Numbers & Math

math

Use Case: Mathematical functions.

import math
print(math.sqrt(16))  # 4.0
print(math.pi)

random

Use Case: Generate random numbers.

import random
print(random.randint(1,10))  # random int between 1-10

statistics

Use Case: Calculate mean, median, variance, etc.

import statistics
data = [1,2,3,4,5]
print(statistics.mean(data))  # 3

Date & Time

datetime

Use Case: Handling dates and times.

from datetime import datetime
now = datetime.now()
print(now.strftime("%Y-%m-%d %H:%M:%S"))

time

Use Case: Time access and delays.

import time
print(time.time())  # current timestamp
time.sleep(1)      # wait for 1 second

Files & OS

os

Use Case: Operating system interfaces.

import os
print(os.getcwd())  # current working directory
os.mkdir("test_folder")

pathlib

Use Case: Object-oriented filesystem paths.

from pathlib import Path
p = Path(".")
print(p.exists())

shutil

Use Case: File copying, moving, and removal.

import shutil
shutil.copy("source.txt", "dest.txt")
